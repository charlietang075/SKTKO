const FLW_URL = 'https://data.weather.gov.hk/weatherAPI/opendata/weather.php?dataType=flw&lang=en'; async function fetchJson(url){ const res = await fetch(url, {cache:'no-cache'}); if(!res.ok) throw new Error('Fetch failed: ' + res.status); return res.json(); } // Generic recursive search: find object with a property matching the areaName function findArea(obj, areaName){ if(!obj || typeof obj !== 'object') return null; if(Array.isArray(obj)){ for(const item of obj){ const found = findArea(item, areaName); if(found) return found; } return null; } // common property names to check const nameKeys = ['name','area','areaName','place','location','district']; for(const k of nameKeys){ if(obj[k] && String(obj[k]).toLowerCase().includes(areaName.toLowerCase())){ return obj; } } for(const k of Object.keys(obj)){ const found = findArea(obj[k], areaName); if(found) return found; } return null; } function renderFlwArea(areaObj){ const container = document.getElementById('forecast'); container.classList.remove('loading'); container.innerHTML = ''; if(!areaObj){ container.textContent = 'Sai Kung forecast not found in dataset.'; return; } // Try to find an array of days in common keys const candidates = ['forecast', 'weatherForecast', 'forecastPeriod', 'data', 'days']; let days = null; for(const k of candidates){ if(Array.isArray(areaObj[k])) { days = areaObj[k]; break; } } // fallback: if areaObj itself is an array if(!days && Array.isArray(areaObj)) days = areaObj; if(!days || !days.length){ container.textContent = 'No forecast days found for Sai Kung.'; return; } days.slice(0,7).forEach(d => { const date = d.forecastDate ?? d.date ?? d.day ?? d.time ?? '—'; const hi = d.maxTemp ?? d.maxTempC ?? d.temperatureMax ?? '—'; const lo = d.minTemp ?? d.minTempC ?? d.temperatureMin ?? '—'; const desc = d.forecast ?? d.forecastWeather ?? d.forecastDesc ?? d.weather ?? '—'; const el = document.createElement('div'); el.className = 'day'; el.innerHTML = `
${date}
${hi}° / ${lo}°
${desc}
`; container.appendChild(el); }); } async function loadSaiKungForecast(){ try{ const json = await fetchJson(FLW_URL); console.log('FLW JSON (inspect):', json); // inspect in console to adapt mapping const area = findArea(json, 'Sai Kung'); renderFlwArea(area); } catch(err){ console.error(err); document.getElementById('forecast').textContent = 'Error loading forecast.'; } } loadSaiKungForecast();
